##3. 系统硬件监控

查看cpu信息
cat /proc/cpuinfo
监听80端口与某机器的通信
sudo ngrep -d eth1 -W byline host remote_ip and port 80

监控进程卡在哪里 perf top -p pid
优化加载动态库：LD_PRELOAD

解决undefined symbol方法：
先 nm -d rados.so |grep symbol 确认so中有该符号
ldd rados.so 找出依赖路径
nm -d 依赖.so |grep symbol

####查看coredump在哪里：
/var/log/message

####abrt扑捉core dump：
1.修改 /etc/abrt.conf ,/etc/abrt-action processpack 为yes
2.service restart

####查看内存：
vmstat 1
格式化：
umount
mkfs
mount
增加用户：

##先切换到root账户，然后
sudo useradd -m wangwenduo
sudo passwd wangwenduo
增加sudo权限：
chmod u+w /etc/sudoers
vim /etc/sudoers
chmod u-x /etc/sudoers
查看设备的使用者
fuser lsof
查看文件系统
df -ihT
删除海量小文件
mkdir empty
rsync -a –delete empty/ your_folder/

统计行数 wc find |xargs wc -l
清空cache和buffer sync; echo 3 > /proc/sys/vm/drop_caches
显示依赖 ldd
更改软链 ln
用显示某些进程并杀掉 ps aux|grep bada| awk ‘{print $2}’ |xargs kill -9
Du -sh 文件大小
Df 查看磁盘
防火墙：
sudo iptables -A INPUT -i {iface} -p tcp -s {ip-address}/{netmask} –dport 6789 -j ACCEPT
sudo iptables -A INPUT -i {iface} -p tcp -s {ip-address}/{netmask} –dport 6800:7300 -j ACCEPT
禁掉某端口的input ：
sudo iptables -A INPUT -p tcp –dport 41977 -j REJECT
显示所有规则：
sudo iptables -L
删除某规则：
sudo iptables -D INPUT 1

限制带宽
yum -y install iproute
/sbin/tc qdisc add dev eth0 root tbf rate 50kbit latency 50ms burst 1000
取消限制：
/sbin/tc qdisc del dev eth0 root tbf

查看网络流量
iftop
dstat

查看硬盘速度：

fio -ioengine=libaio -bs=4k -direct=1 -thread -rw=randread -size=40G -filename=test -name=”360 4k random read” -iodepth=8 -runtime=60

查看自己的程序：
top -U $USER

查看父进程
ps -ef
查看所有句柄
lsof -n | awk ‘{print $2}’ | sort | uniq -c | sort -nr | awk ‘{print $2}’|more

以格式查看json
jq
